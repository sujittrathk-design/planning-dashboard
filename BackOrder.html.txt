<script>
async function load(){
  const data = await fetchData("backorder");

  let html = "<tr><th>SKU</th><th>Stock</th></tr>";

  data.slice(1).forEach(r=>{
    html += `<tr>
      <td>${r[3]}</td>
      <td>${r[7]}</td>
    </tr>`;
  });

  document.getElementById("table").innerHTML = html;
}
load();
</script>
